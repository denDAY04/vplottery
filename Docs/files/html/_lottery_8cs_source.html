79  public Lottery(int lotteryNumberLimit, int ticketPrice, List<Ticket> existingTickets = null, List<int> winningLottryNumbers = null ) {
80  if (lotteryNumberLimit < LotteryNumberMin)
81  throw new ArgumentException(InvalidLotteryNumberErrorMsg);
82  if (ticketPrice < 0)
83  throw new ArgumentException(InvalidTicketPriceErrorMsg);
84 
85  LotteryNumberMax = lotteryNumberLimit;
86  TicketPrice = ticketPrice;
87  WinningLotteryNumbers = winningLottryNumbers ?? new List<int>();
88  Rng = new Random();
89  NumLotteryNumbersLeft = LotteryNumberMax;
90 
91  // Flag unused numbers and then store the lottery number-range.
92  LotteryNumbers = new List<int>(LotteryNumberMin + LotteryNumberMax);
93  for (var unusedPos = 0; unusedPos < LotteryNumberMin; unusedPos++){
94  LotteryNumbers.Add(InvalidLotteryNumber);
95  }
96  for (var lotteryNumber = LotteryNumberMin; lotteryNumber <= LotteryNumberMax; lotteryNumber++) {
97  LotteryNumbers.Add(lotteryNumber);
98  }
99 
100  // If existing session, loads tickets and remove their numbers from the availible number range.
101  if (existingTickets != null) {
102  Tickets = existingTickets;
103  foreach (var ticket in Tickets) {
104  LotteryNumbers[ticket.LotteryNumber] = InvalidLotteryNumber;
105  --NumLotteryNumbersLeft;
106  }
107  } else {
108  Tickets = new List<Ticket>();
109  }
110  }

130  public Ticket DrawWinningNumber() {
131  // If there are no sold tickets, don't draw a number.
132  if (Tickets.Count < 1)
133  return null;
134  // If every sold ticket has won, don't draw a number.
135  if (WinningLotteryNumbers.Count == Tickets.Count)
136  return null;
137 
138  // Pick a winning number, checking that it's not still availibe
139  // (i.e. a ticket has this number) and it's not already been picked.
140  var winningNumber = Rng.Next(LotteryNumberMin, LotteryNumberMax + 1);
141  while (LotteryNumbers[winningNumber] != InvalidLotteryNumber || WinningLotteryNumbers.Contains(winningNumber)){
142  winningNumber = Rng.Next(LotteryNumberMin, LotteryNumberMax + 1);
143  }
144 
145  // Store winning number and return the winning ticket.
146  WinningLotteryNumbers.Add(winningNumber);
147  return Tickets.Find(T => T.LotteryNumber == winningNumber);
148  }

161  public LotteryNumberSaleResult BuyLotteryNumber(int lotteryNumber, string buyer)

184  public int BuyLotteryNumber(string buyer) {
185  if (NumLotteryNumbersLeft < 1)
186  return InvalidLotteryNumber;
187 
188  // Pick number and try and buy it.
189  var lotteryNumber = Rng.Next(LotteryNumberMin, LotteryNumberMax) + 1;
190  var result = BuyLotteryNumber(lotteryNumber, buyer);
191  while (result == LotteryNumberSaleResult.NumberAlreadySold) {
192  lotteryNumber = Rng.Next(LotteryNumberMin, LotteryNumberMax + 1);
193  result = BuyLotteryNumber(lotteryNumber, buyer);
194  }
195 
196  if (result == LotteryNumberSaleResult.NumberOutOfRange)
197  throw new ArgumentException("Internal error for random-bought ticket: number was out of the lottery range.");
198 
199  return lotteryNumber;
200  }

303  public static ImportResult ImportFromFile(Stream xmlImportFIle, out Lottery importedLottery, out string parseErrorMsg) {

305  importedLottery = null;
306  parseErrorMsg = "";

308  // Check for root tag.
309  if (xmlRoot == null) {
310  return ImportResult.InvalidFile;
311  }
312  if (xmlRoot.Name != XmlTagNames.Root) {
313  parseErrorMsg = "Missing root tag " + XmlTagNames.Root;
314  return ImportResult.ParseError;
315  }
316 
317  // Parse data
318 
319  // MaxLotteryNumber
320  int maxLotteryNumber;
321  var node = xmlRoot.Descendants(XmlTagNames.MaxLotteryNumber).FirstOrDefault();
322  if (node == null) {
323  parseErrorMsg = "Missing tag " + XmlTagNames.MaxLotteryNumber;
324  return ImportResult.ParseError;
325  }
326  try {
327  maxLotteryNumber = int.Parse(node.Value);
328  } catch (FormatException) {
329  parseErrorMsg = "Malformed data in tag " + XmlTagNames.MaxLotteryNumber;
330  return ImportResult.ParseError;
331  }
332

334  var tickets = new List<Ticket>();
335  node = xmlRoot.Descendants(XmlTagNames.Tickets).FirstOrDefault();
336  if (node == null) {
337  parseErrorMsg = "Missing tag " + XmlTagNames.Tickets;
338  return ImportResult.ParseError;
339  }
340 
341  // Ticket price attribute
342  int ticketPrice;
343  var priceAttribute = node.Attribute(XmlTagNames.TicketsPriceAttribute);
344  if (priceAttribute == null) {
345  parseErrorMsg = "Missing " + XmlTagNames.TicketsPriceAttribute + " attribute in tag" +
346  XmlTagNames.Tickets;
347  return ImportResult.ParseError;
348  }
349  try {
350  ticketPrice = int.Parse(priceAttribute.Value);
351  } catch (FormatException) {
352  parseErrorMsg = "Malformed data in attribute " + XmlTagNames.TicketsPriceAttribute + " of tag " +
353  XmlTagNames.Tickets;
354  return ImportResult.ParseError;
355  }
356 
357  // Sequence of tickets - Optional
358  if (node.Descendants(XmlTagNames.Ticket).Any()) {
359  foreach (var ticket in node.Descendants(XmlTagNames.Ticket)) {
360  // Mandatory lottery number
361  Ticket parsedTicket;

363  if (lotteryNumberNode == null) {

370  } catch (FormatException) {

372  return ImportResult.ParseError;
373  }
374 
375  // Optional owner
376  var ownerNode = ticket.Descendants(XmlTagNames.TicketOwner).FirstOrDefault();
377  if (ownerNode != null)

379 
380  // Optional comment

382  if (commentNode != null)

385  tickets.Add(parsedTicket);
386  }
387  }

392  if ( node != null && node.Descendants(XmlTagNames.WinningLotteryNumber).Any()) {
393  foreach (var winner in node.Descendants(XmlTagNames.WinningLotteryNumber)) {
394  // Mandatory lottery number attribute
395  var lotteryNumberAttribute = winner.Attribute(XmlTagNames.WinningLotteryNumberAttribute);
396  if (lotteryNumberAttribute == null) {

398  XmlTagNames.WinningLotteryNumber;

405  return ImportResult.ParseError;
406  }
407  }
408  }
409
